# 입, 출력 기본

```c++
#include <iostream>

using namespace std;

int main() {
  int a, b, c, d, e;

  cin >> a >> b >> c >> d >> e; // 입력

  cout << a << c; // 출력

  cout << a << " " << c << " " << 5; // 공백

}

```

```c++
#include <iostream>

using namespace std;

int main() {
  int a, b;

  cin >> a >> b; // 입력

  cout << a - b; // 출력

  cout << '\n' // 개행

}

```
